set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_LIBRARY_OUTPUT_DIRECTORY})

#################################
set( VDB_LIB_RD "" )
foreach( target ${VDB_LIB_RD_TARGETS})
    list( APPEND VDB_LIB_RD "$<TARGET_OBJECTS:${target}>" )
endforeach()

add_library( ncbi-vdb STATIC "${VDB_LIB_RD}" )
ExportStatic( ncbi-vdb true )

set( SHARED_SRC_DIR ${CMAKE_INSTALL_PREFIX}/lib64/vdb_shared_sources )

message("Installing ncbi-vdb interfaces (headers) into ${CMAKE_INSTALL_PREFIX}/lib64")
message("Installing ncbi-vdb shared sources into ${SHARED_SRC_DIR}")

install( DIRECTORY ${CMAKE_SOURCE_DIR}/interfaces DESTINATION ${CMAKE_INSTALL_PREFIX}/lib64 )
install( FILES
	${CMAKE_SOURCE_DIR}/libs/vdb/blob-priv.h
	${CMAKE_SOURCE_DIR}/libs/vdb/page-map.h
	DESTINATION ${SHARED_SRC_DIR}/libs/vdb )
install( FILES
	${CMAKE_SOURCE_DIR}/libs/kfg/default.kfg
	DESTINATION ${SHARED_SRC_DIR}/libs/kfg )
install( FILES
	${CMAKE_SOURCE_DIR}/libs/align/samextract-pool.h
	${CMAKE_SOURCE_DIR}/libs/align/samextract.h
	DESTINATION ${SHARED_SRC_DIR}/libs/align )
install( FILES
	${CMAKE_SOURCE_DIR}/libs/klib/release-vers.h
	DESTINATION ${SHARED_SRC_DIR}/libs/klib )
install( FILES
	${CMAKE_SOURCE_DIR}/libs/vfs/json-response.h
	${CMAKE_SOURCE_DIR}/libs/vfs/services-priv.h
	DESTINATION ${SHARED_SRC_DIR}/libs/vfs )

# install( TARGETS ncbi-vdb DESTINATION ${CMAKE_INSTALL_PREFIX}/lib64 EXPORT FindNcbiVdb)
# install( EXPORT FindNcbiVdb DESTINATION ${CMAKE_INSTALL_PREFIX}/lib64 )
MSVS_StaticRuntime( ncbi-vdb )

    ${VDB_LIB_RD}
    # $<TARGET_OBJECTS:ngs>
    # $<TARGET_OBJECTS:ngs-jni>
    # $<TARGET_OBJECTS:ngs-py>
)

add_library( ncbi-vdb-shared SHARED "${VDB_LIB_RD_SHARED}" )
set_target_properties( ncbi-vdb-shared PROPERTIES OUTPUT_NAME ncbi-vdb )
#target_link_libraries( ncbi-vdb-shared ncbi-ngs ${LIBXML2} m dl pthread )
target_link_libraries( ncbi-vdb-shared ${LIBXML2} m dl pthread )
MakeLinksShared( ncbi-vdb-shared ncbi-vdb true )
