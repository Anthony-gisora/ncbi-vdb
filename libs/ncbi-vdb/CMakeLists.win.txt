set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY_DEBUG ${CMAKE_LIBRARY_OUTPUT_DIRECTORY_DEBUG})
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY_RELEASE ${CMAKE_LIBRARY_OUTPUT_DIRECTORY_RELEASE})

#############################
set( VDB_LIB_RD "" )
foreach( target ${VDB_LIB_RD_TARGETS})
    list( APPEND VDB_LIB_RD "$<TARGET_OBJECTS:${target}>" )
endforeach()

add_library( ncbi-vdb STATIC "${VDB_LIB_RD}" )
MSVS_StaticRuntime( ncbi-vdb )
ExportStatic( ncbi-vdb true )
ExportShared( ncbi-vdb true )
target_link_libraries( ncbi-vdb-shared ${LIBXML2} Crypt32 )
target_link_libraries( ncbi-vdb-shared-md ${LIBXML2} Crypt32 )

#################################
set( VDB_LIB_RDWR "" )
foreach( target ${VDB_LIB_RDWR_TARGETS})
    list( APPEND VDB_LIB_RDWR "$<TARGET_OBJECTS:${target}>" )
endforeach()

add_library( ncbi-wvdb STATIC "${VDB_LIB_RDWR}" )
MSVS_StaticRuntime( ncbi-wvdb )
ExportStatic( ncbi-wvdb true )
ExportShared( ncbi-wvdb true )
target_link_libraries( ncbi-wvdb-shared ${LIBXML2} Crypt32 )
target_link_libraries( ncbi-wvdb-shared-md ${LIBXML2} Crypt32 )

################################
# additional files to install

set( SHARED_SRC_DIR ${CMAKE_INSTALL_PREFIX}/vdb_shared_sources )

message("Installing ncbi-vdb interfaces (headers) into ${CMAKE_INSTALL_PREFIX}")
message("Installing ncbi-vdb shared sources into ${SHARED_SRC_DIR}")

install( DIRECTORY ${CMAKE_SOURCE_DIR}/interfaces DESTINATION ${CMAKE_INSTALL_PREFIX} )
install( FILES
	${CMAKE_SOURCE_DIR}/libs/vdb/blob-priv.h
	${CMAKE_SOURCE_DIR}/libs/vdb/page-map.h
	DESTINATION ${SHARED_SRC_DIR}/libs/vdb )
install( FILES
	${CMAKE_SOURCE_DIR}/libs/kfg/default.kfg
	DESTINATION ${SHARED_SRC_DIR}/libs/kfg )
install( FILES
	${CMAKE_SOURCE_DIR}/libs/align/samextract-pool.h
	${CMAKE_SOURCE_DIR}/libs/align/samextract.h
	DESTINATION ${SHARED_SRC_DIR}/libs/align )
install( FILES
	${CMAKE_SOURCE_DIR}/libs/klib/release-vers.h
	DESTINATION ${SHARED_SRC_DIR}/libs/klib )
install( FILES
	${CMAKE_SOURCE_DIR}/libs/vfs/json-response.h
	${CMAKE_SOURCE_DIR}/libs/vfs/services-priv.h
	DESTINATION ${SHARED_SRC_DIR}/libs/vfs )
