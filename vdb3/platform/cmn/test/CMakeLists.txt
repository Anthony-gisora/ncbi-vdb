include_directories( .. )
add_definitions( -DTESTING )

function( test stem )
    add_executable ( test-cmn3-${stem} ${stem}-tst.cpp )
    target_link_libraries ( test-cmn3-${stem} gtest kfc3 utf8proc ${LQUADMATH} )
    add_test ( NAME Test_CMN3_${stem} COMMAND test-cmn3-${stem} )
endfunction()
